# Pull the latest MySQL image
docker pull mysql:latest

# Run a MySQL container named test-mysql with specified environment variables and port mapping
docker run --name test-mysql -e MYSQL_ROOT_PASSWORD=pass -e MYSQL_USER=user -e MYSQL_PASSWORD=test -p 3307:3306 -d mysql

# Access the MySQL container
docker exec -it test-mysql mysql -u root -p

# Enter the MySQL root password when prompted
mysql -u root -p

# Once inside the MySQL shell, grant privileges and flush them
GRANT ALL PRIVILEGES ON *.* TO 'user'@'%' WITH GRANT OPTION;
FLUSH PRIVILEGES;

CREATE DATABASE `test-mysql`;
